### `parse_psv0.py` (the ad-hoc PSV0 header reader used above)

```python
import sys, struct
d = open(sys.argv[1], "rb").read()
assert d[:4] == b"DXBC"
partcount = struct.unpack_from("<I", d, 28)[0]
offs = [struct.unpack_from("<I", d, 32 + 4 * i)[0] for i in range(partcount)]
parts = []
for o in offs:
    name = d[o:o+4].decode("latin1")
    parts.append(name)
    if name == "PSV0":
        print("  PSV0 InfoSize (first u32) =", struct.unpack_from("<I", d, o + 8)[0])
print("  parts:", parts)
```
